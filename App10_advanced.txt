        function runController(app)

            % ----- 1. Housekeeping -----
            app.StopFlag = 0;
            if ishandle(app.PlotFigure)
                close(app.PlotFigure);
                app.PlotFigure = [];
            end
            clc;

            % ----- 2. Constants -----
            IntErrorMax       =  15;
            IntErrorMin       = -15;
            R_Series          = 5100;
            Vcc               = 5;
            DataVectorSizeMax = 10000;

            % ----- 3. Read averaging samples -----
            NrSamples      = app.NrSamplesSpinner.Value;
            AverageVoltage = zeros(NrSamples, 1);

            % ----- 4. Auto‑calibration -----
            writePWMVoltage(app.a, 'D3', 5);
            for i = 1:NrSamples
                AverageVoltage(i) = readVoltage(app.a, 'A0');
            end
            SetPointMin = mean(AverageVoltage);

            writePWMVoltage(app.a, 'D3', 0);
            for i = 1:NrSamples
                AverageVoltage(i) = readVoltage(app.a, 'A0');
            end
            SetPointMax = mean(AverageVoltage);

            % ----- 5. Apply calibrated limits -----
            minLim = round(SetPointMin, 1);
            maxLim = round(SetPointMax, 1);
            app.ActualValueGauge.Limits  = [minLim, maxLim];
            app.SetPointSlider.Limits    = [minLim, maxLim];
            app.SetPointEditField.Limits = [minLim, maxLim];

            % Update slider ticks to match calibrated range
            app.SetPointSlider.MajorTicks = linspace(minLim, maxLim, 5);
            tickLabels = arrayfun(@(x) sprintf('%.1f', x), app.SetPointSlider.MajorTicks, ...
                                  'UniformOutput', false);
            app.SetPointSlider.MajorTickLabels = tickLabels;

            % ----- 6. Initial setpoint = midpoint -----
            midPoint = (minLim + maxLim) / 2;
            app.SetPointEditField.Value = midPoint;
            app.SetPointSlider.Value    = midPoint;

            % ----- 7. Reset GUI displays -----
            app.ActualValueEditField.Value = 0;
            app.ErrorEditField.Value       = 0;
            app.ErrorIntEditField.Value    = 0;
            app.ErrorDifEditField.Value    = 0;
            app.PWMoutputEditField.Value   = 0;
            app.R_LDREditField.Value       = 0;
            app.ActualValueGauge.Value     = minLim;
            app.ErrorGauge.Value           = 0;
            app.ErrorIntGauge.Value        = 0;
            app.ErrorDifGauge.Value        = 0;
            app.PWMoutputGauge.Value       = 0;
            app.R_LDRGauge.Value           = 0;

            % ----- 8. Pre‑allocate data vectors (including TimeVec) -----
            app.ActualValueVec   = zeros(DataVectorSizeMax, 1);
            app.SetPointVec      = zeros(DataVectorSizeMax, 1);
            app.ErrorVec         = zeros(DataVectorSizeMax, 1);
            app.ErrorIntVec      = zeros(DataVectorSizeMax, 1);
            app.ErrorDifVec      = zeros(DataVectorSizeMax, 1);
            app.OutputVoltageVec = zeros(DataVectorSizeMax, 1);
            app.TimeVec          = zeros(DataVectorSizeMax, 1);   % FIXED
            app.KpVec            = zeros(DataVectorSizeMax, 1);
            app.KiVec            = zeros(DataVectorSizeMax, 1);
            app.KdVec            = zeros(DataVectorSizeMax, 1);
            app.CurrentN         = 1;

            % ----- Reset gain change log and applied gains -----
            app.GainChangeLog = {};
            app.LastKp = app.KpValueSlider.Value;
            app.LastKi = app.KiValueSlider.Value;
            app.LastKd = app.KdValueSlider.Value;
            app.Kp_applied = app.KpValueSlider.Value;
            app.Ki_applied = app.KiValueSlider.Value;
            app.Kd_applied = app.KdValueSlider.Value;
            app.ElapsedTime = 0;
            app.FilteredPV = NaN;

            % Reset switch to OFF at start
            app.GainsChangingSwitch.Value = 'Off';

            % ----- PID state variables -----
            Errorint = 0;
            ActualValueOld = NaN;
            n = 1;
            loopTimer = tic;

            % ----- 9. Create live plot figure -----
            app.PlotFigure = figure( ...
                'Name',     'PID Live Plots', ...
                'Units',    'normalized', ...
                'Position', [0.55 0.05 0.43 0.88]);

            subplot(5,1,1);
            hAV = plot(nan, 'b-', 'LineWidth', 1.2); hold on
            hSP = plot(nan, 'r--', 'LineWidth', 1.2);
            ylabel('V'); title('Actual Value vs Setpoint, V'); grid on
            legend('Actual Value','SetPoint','Location','best')
            ylim([minLim, maxLim])

            subplot(5,1,2);
            hER = plot(nan, 'Color', [0.2 0.6 0.2], 'LineWidth', 1);
            ylabel('V'); title('Error, V'); grid on

            subplot(5,1,3);
            hEI = plot(nan, 'Color', [0.8 0.4 0], 'LineWidth', 1);
            ylabel('Accum V'); title('Integrated Error, V'); grid on

            subplot(5,1,4);
            hED = plot(nan, 'Color', [0.5 0 0.5], 'LineWidth', 1);
            ylabel('V/s'); title('Derivative (dPV/dt), V/s'); grid on

            subplot(5,1,5);
            hOV = plot(nan, 'r-', 'LineWidth', 1);
            ylabel('V'); title('PWM Output Voltage, V'); grid on
            ylim([0 5])

           % ----- 10. PID CONTROL LOOP -----
            while isvalid(app) && isprop(app,'StopFlag') && app.StopFlag == 0

                % ---- Bounds check (prevent writing beyond allocated size) ----
                if n > DataVectorSizeMax
                    app.StopFlag = 1;
                    uialert(app.UIFigure, ...
                        'Data limit reached (10,000 samples). Controller stopped. Save CSV.', ...
                        'Memory Limit Reached');
                    break;
                end

                % ---- Pause handling (LED off while paused) ----
                if app.PauseFlag == 1
                    writePWMVoltage(app.a, 'D3', 0);   % LED off
                    pause(0.05);
                    drawnow;
                    loopTimer = tic;                   % reset timer to avoid dt jump
                    continue;
                end

                % ---- Measure real loop time (dt) ----
                dt = toc(loopTimer);
                if dt <= 0; dt = 0.001; end
                loopTimer = tic;
                app.ElapsedTime = app.ElapsedTime + dt;
                app.TimeVec(n) = app.ElapsedTime;      % store real time

                % ---- (a) Setpoint ----
                SetPoint = app.SetPointSlider.Value;
                app.SetPointEditField.Value = SetPoint;
                app.SetPointVec(n) = SetPoint;

                % ---- (b) Read and average LDR voltage ----
                for i = 1:NrSamples
                    AverageVoltage(i) = readVoltage(app.a, 'A0');
                end
                ActualValue = mean(AverageVoltage);
                app.ActualValueGauge.Value = ActualValue;
                app.ActualValueEditField.Value = ActualValue;
                app.ActualValueVec(n) = ActualValue;

                % ---- (c.0) LDR resistance (display only) ----
                R_LDR = R_Series * (ActualValue / (Vcc - ActualValue));
                app.R_LDRGauge.Value = R_LDR;
                app.R_LDREditField.Value = R_LDR;

                % ---- (c.1) Proportional error ----
                Error = SetPoint - ActualValue;
                app.ErrorGauge.Value = Error;
                app.ErrorEditField.Value = Error;
                app.ErrorVec(n) = Error;

                % ---- (c.2) Integral with anti‑windup (scaled by dt) ----
                Errorint = Errorint + (Error * dt);
                if Errorint > IntErrorMax; Errorint = IntErrorMax; end
                if Errorint < IntErrorMin; Errorint = IntErrorMin; end
                app.ErrorIntGauge.Value = Errorint;
                app.ErrorIntEditField.Value = Errorint;
                app.ErrorIntVec(n) = Errorint;

                % ---- (c.3) Derivative of Process Variable (filtered) ----
                % Exponential moving average to reduce noise
                if isnan(app.FilteredPV)
                    app.FilteredPV = ActualValue;
                else
                    app.FilteredPV = 0.8 * app.FilteredPV + 0.2 * ActualValue;
                end

                if isnan(ActualValueOld)
                    ActualValueOld = app.FilteredPV;
                end
                DerivativeTerm = (app.FilteredPV - ActualValueOld) / dt;

                app.ErrorDifGauge.Value = DerivativeTerm;
                app.ErrorDifEditField.Value = DerivativeTerm;
                app.ErrorDifVec(n) = DerivativeTerm;

                % ---- (d) Read applied gains (updated by slider callbacks) ----
                Kp = app.Kp_applied;
                Ki = app.Ki_applied;
                Kd = app.Kd_applied;

                % Record per‑sample gains (always)
                app.KpVec(n) = Kp;
                app.KiVec(n) = Ki;
                app.KdVec(n) = Kd;

                % ---- Log gain changes using real time ----
                t = app.ElapsedTime;
                if abs(Kp - app.LastKp) > 0.001
                    app.GainChangeLog(end+1,:) = {n, t, 'Kp', app.LastKp, Kp, ...
                        sprintf('Kp: %.3f → %.3f', app.LastKp, Kp)};
                    app.LastKp = Kp;
                end
                if abs(Ki - app.LastKi) > 0.001
                    app.GainChangeLog(end+1,:) = {n, t, 'Ki', app.LastKi, Ki, ...
                        sprintf('Ki: %.3f → %.3f', app.LastKi, Ki)};
                    app.LastKi = Ki;
                end
                if abs(Kd - app.LastKd) > 0.001
                    app.GainChangeLog(end+1,:) = {n, t, 'Kd', app.LastKd, Kd, ...
                        sprintf('Kd: %.3f → %.3f', app.LastKd, Kd)};
                    app.LastKd = Kd;
                end

                % ---- (e) PID output (negated for inverted plant) ----
                OutputVoltage = -(Kp * Error + Ki * Errorint - Kd * DerivativeTerm);
                OutputVoltage = max(0, min(5, OutputVoltage));   % clamp to [0,5]

                % ---- (f) Send to LED with hardware error handling ----
                try
                    writePWMVoltage(app.a, 'D3', OutputVoltage);
                catch
                    app.StopFlag = 1;
                    uialert(app.UIFigure, ...
                        'Hardware connection lost. Controller stopped.', ...
                        'Hardware Error');
                    break;
                end

                app.PWMoutputGauge.Value = OutputVoltage;
                app.PWMoutputEditField.Value = OutputVoltage;
                app.OutputVoltageVec(n) = OutputVoltage;

                % ---- (g) Save filtered PV for next derivative ----
                ActualValueOld = app.FilteredPV;

                % ---- (h) Update live plots (throttled) ----
                if mod(n, 1) == 0
                    xData = 1:n;
                    set(hAV, 'XData', xData, 'YData', app.ActualValueVec(1:n)');
                    set(hSP, 'XData', xData, 'YData', app.SetPointVec(1:n)');
                    set(hER, 'XData', xData, 'YData', app.ErrorVec(1:n)');
                    set(hEI, 'XData', xData, 'YData', app.ErrorIntVec(1:n)');
                    set(hED, 'XData', xData, 'YData', app.ErrorDifVec(1:n)');
                    set(hOV, 'XData', xData, 'YData', app.OutputVoltageVec(1:n)');
                    drawnow;
                end

                % ---- (i) Increment index and pause for GUI responsiveness ----
                n = n + 1;
                app.CurrentN = n;
                pause(0.01);
            end

        end

        % Button pushed function: STOPButton
        function STOPButtonPushed(app, event)
            app.StopFlag = 1;
            pause(0.05);
            bufferCleared = false;
            while ~bufferCleared
                try
                    readVoltage(app.a, 'A0');
                    bufferCleared = true;
                catch
                end
            end
        end

            % action is needed — the new setpoint will be picked up on the
            % next loop iteration
        end

            % so an out-of-range entry is silently clamped.
        end

        % Button pushed function: RESTARTButton
        function RESTARTButtonPushed(app, event)
            app.StopFlag = 1;
            app.PauseFlag = 1;
            pause(0.05);
            bufferCleared = false;
            while ~bufferCleared
                try
                    readVoltage(app.a, 'A0');
                    bufferCleared = true;
                catch
                end
            end
            if ishandle(app.PlotFigure)
                close(app.PlotFigure);
                app.PlotFigure = [];
            end
            runController(app);
        end

        % Button pushed function: SAVECSVButton
        function SAVECSVButtonPushed(app, event)
            nSaved = app.CurrentN - 1;

            if nSaved < 1
                uialert(app.UIFigure, ...
                    'No data recorded yet. Press START first.', ...
                    'Nothing to save');
                return
            end

            timestamp = datestr(now, 'yyyy-mm-dd_HH-MM-SS');
            
            % [CORRECTION NOTE: True Timestamp Export]
            % Now exports the actual tracked array rather than fake dt math
            Time_s    = app.TimeVec(1:nSaved);

            % --- FILE 1: Main data log ------------------------------------
            % Kp, Ki, Kd columns always present regardless of switch state.
            T = table( ...
                (1:nSaved)',                    ...
                Time_s,                         ...
                app.SetPointVec(1:nSaved),      ...
                app.ActualValueVec(1:nSaved),   ...
                app.ErrorVec(1:nSaved),         ...
                app.ErrorIntVec(1:nSaved),      ...
                app.ErrorDifVec(1:nSaved),      ...
                app.OutputVoltageVec(1:nSaved), ...
                app.KpVec(1:nSaved),            ...
                app.KiVec(1:nSaved),            ...
                app.KdVec(1:nSaved),            ...
                'VariableNames', {              ...
                    'Sample',                   ...
                    'Time_s',                   ...
                    'SetPoint_V',               ...
                    'ActualValue_V',            ...
                    'Error_V',                  ...
                    'ErrorInt_V',               ...
                    'ErrorDif_V',               ...
                    'OutputVoltage_V',          ...
                    'Kp',                       ...
                    'Ki',                       ...
                    'Kd'});

            dataFile = ['PID_Data_' timestamp '.csv'];
            writetable(T, dataFile);

            % --- FILE 2: Gain change event log ----------------------------
            % Switch OFF → one row per mouse release (coarse)
            % Switch ON  → one row per 10 ms change while dragging (fine)
            logFile = ['PID_GainLog_' timestamp '.csv'];

            if isempty(app.GainChangeLog)
                TLog = table( ...
                    {'No gain changes detected.'}, ...
                    'VariableNames', {'Note'});
            else
                TLog = cell2table(app.GainChangeLog, ...
                    'VariableNames', {  ...
                        'Sample',       ...
                        'Time_s',       ...
                        'Gain',         ...
                        'OldValue',     ...
                        'NewValue',     ...
                        'Description'});
            end

            writetable(TLog, logFile);

            % --- Confirm to user ------------------------------------------
            uialert(app.UIFigure, ...
                [num2str(nSaved) ' samples saved.' newline newline ...
                 'Data file:'  newline dataFile  newline newline ...
                 'Gain log:'   newline logFile], ...
                'CSV Saved');
        end
